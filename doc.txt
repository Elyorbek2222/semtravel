const WP_API_URL = 'https://api.example.com/wp-json/wp/v2';


export interface Post {
  id: number;
  slug: string;
  title: { rendered: string };
  content: { rendered: string };
  excerpt: { rendered: string };
  date: string;
  modified: string;
  featured_image_url: string;
  reading_time: number;
  yoast_head_json: {
    title: string;
    description: string;
    og_image: string[];
    schema: any;
  };
  categories: number[];
  tags: number[];
}


export async function getPosts(params = {}): Promise<Post[]> {
  const searchParams = new URLSearchParams(params);
  const res = await fetch(`${WP_API_URL}/posts?${searchParams}`, {
    next: { revalidate: 3600 } // ISR: revalidate every hour
  });
  
  if (!res.ok) throw new Error('Failed to fetch posts');
  return res.json();
}


export async function getPost(slug: string): Promise<Post> {
  const res = await fetch(`${WP_API_URL}/posts?slug=${slug}`, {
    next: { revalidate: 60 } // ISR: revalidate every minute
  });
  
  if (!res.ok) throw new Error('Failed to fetch post');
  const posts = await res.json();
  return posts[0];
}


export async function getCategories() {
  const res = await fetch(`${WP_API_URL}/categories`, {
    next: { revalidate: 86400 } // Revalidate once a day
  });
  
  if (!res.ok) throw new Error('Failed to fetch categories');
  return res.json();
}

export async function generateMetadata({ params }): Promise<Metadata> {

    },
  };
}

4. app/sitemap.xml/route.ts (Динамический sitemap):
typescript
import { getPosts } from '@/lib/wordpress';


export async function GET() {
  const posts = await getPosts({ per_page: 100 });
  
  const sitemap = `<?xml version="1.0" encoding="UTF-8"?>
    <urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
      <url>
        <loc>https://example.com</loc>
        <lastmod>${new Date().toISOString()}</lastmod>
        hangefreq>daily</changefreq>
        <priority>1.0</priority>
      </url>
      ${posts.map((post) => `
        <url>
          <loc>https://example.com/${post.slug}</loc>
          <lastmod>${post.modified}</lastmod>
          hangefreq>weekly</changefreq>
          <priority>0.8</priority>
        </url>
      `).join('')}
    </urlset>`;
  
  return new Response(sitemap, {
    headers: {
      'Content-Type': 'application/xml',
    },
  });
}

НОДА 12: Compile Final Article (Code Node)
Код для сборки финального HTML:
javascript
const outline = $json.outline;
const sections = $json.sections; // все написанные секции
const images = $json.images || []; // если есть
const faq = outline.faq;


// Собрать full HTML
let html = `
<!-- Intro -->
<p>${outline.intro}</p>


<!-- Table of Contents -->
<h2>Содержание</h2>
<ul>
${outline.sections.map((s, i) => `<li><a href="#section-${i}">${s.h2}</a></li>`).join('\n')}
</ul>


<!-- Основной контент -->
${sections.map((section, i) => `
<section id="section-${i}">
  ${section.html_content}
  ${images[i] ? `
    <figure>
      <img src="${images[i].url}" alt="${images[i].alt}" width="800" height="600" loading="lazy">
      <figcaption>${images[i].caption}</figcaption>
    </figure>
  ` : ''}
</section>
`).join('\n')}


<!-- FAQ -->
<h2>Часто задаваемые вопросы</h2>
${faq.map(q => `
<div class="faq-item">
  <h3>${q.question}</h3>
  <p>${q.short_answer}</p>
</div>
`).join('\n')}


<!-- Заключение -->
<h2>Итог</h2>
<p>${outline.conclusion}</p>
<p><strong>Ваш следующий шаг:</strong> ${outline.cta}</p>
`;


// Подготовить данные для WordPress
return {
  json: {
    title: outline.h1,
    content: html,
    excerpt: outline.meta_description,
    slug: outline.slug,
    status: 'publish',
    meta: {
      description: outline.meta_description,
      keywords: $json.related_keywords.join(', '),
    },
    featured_media_id: images[0]?.media_id || null,
    categories: [1], // ID категории по умолчанию
    tags: [], // tags по необходимости
  }
};

✅ 1.1 Meta Tags (в <head> секции)
Что должно быть
В HTML коде страницы в секции <head> должны присутствовать следующие теги:
xml
<title>Заголовок статьи 55-60 символов с ключевым словом</title>
<meta name="description" content="Описание 155-160 символов с ключевым словом и призывом">
<link rel="canonical" href="https://example.com/url-статьи">
<meta property="og:title" content="Заголовок для соцсетей">
<meta property="og:description" content="Описание для соцсетей">
<meta property="og:image" content="https://example.com/image.jpg">
<meta property="og:type" content="article">
<meta name="twitter:card" content="summary_large_image">


Как проверить
Способ 1: Ручная проверка в коде
1. Открыть статью в браузере
2. Правая кнопка мыши → "Просмотр кода страницы" (View Page Source) или нажать Ctrl+U
3. Нажать Ctrl+F (поиск по странице)
4. Последовательно искать каждый тег:
   * Искать: <title> → Должен быть найден РОВНО 1 раз
   * Искать: <meta name="description" → Должен быть найден РОВНО 1 раз
   * Искать: <link rel="canonical" → Должен быть найден РОВНО 1 раз
   * Искать: <meta property="og: → Должно быть найдено минимум 4 раза (title, description, image, type)
Способ 2: Использование расширения (РЕКОМЕНДУЕТСЯ)
1. Установить расширение для браузера: "SEO Meta in 1 Click"
   * Chrome: https://chrome.google.com/webstore → искать "SEO Meta in 1 Click"
   * Firefox: https://addons.mozilla.org → искать "SEO Meta in 1 Click"
2. Открыть статью на сайте
3. Нажать на иконку расширения в панели браузера
4. Посмотреть вкладки:
   * Summary — общая информация о странице
   * OpenGraph — теги для соцсетей

✅ 1.4 Semantic HTML (семантическая разметка)
Что должно быть

  </header>


  <section id="intro">
    <p>Вводный параграф...</p>
  </section>


  <section id="section-1">
    <h2>Первый раздел</h2>
    <p>Контент раздела...</p>
  </section>


  <section id="section-2">
    <h2>Второй раздел</h2>
    <p>Контент раздела...</p>
  </section>
  
  <footer>
    <p>Информация об авторе...</p>
  </footer>
</article>

* ✅ Вся статья обернута в тег <article>
* ✅ Разделы используют <section> вместо <div>
* ✅ Заголовок статьи в <header> теге
* ✅ Футер статьи (автор, дата) в <footer> теге
* ✅ Дата публикации в <time> теге с атрибутом datetime

5. Ctrl+F → искать <header> → Должно быть найдено (для заголовка статьи)

✅ 1.6 Структура URL (человекочитаемый адрес)
Что должно быть:
   * URL короткий, логичный и понятный человеку
   * Содержит главное ключевое слово
   * Только латиница, цифры и дефисы

/kak_zarabotat_na_wb?ref=123
Как проверить:
      1. Скопировать URL из адресной строки
      2. Проверить визуально:
      * Понятно ли по одному URL, о чем статья?
      * Есть ли там ключевое слово?
      * Нет ли русских букв, лишних чисел, параметров типа ?id=?
Что замерять:
text
URL: _________________________________
Содержит ключевое слово: ДА / НЕТ
Только латиница и дефисы: ДА / НЕТ
Без лишних параметров ( ? , = , & ): ДА / НЕТ
Длина URL < 75 символов: ДА / НЕТ
Понятно по URL, о чем статья: ДА / НЕТ


Требования к приемке:
      * ✅ В URL есть основное ключевое слово
      * ✅ Нет кириллицы, спецсимволов, параметров (?, &, =)

✅ 2.2 Answer-First Structure (прямые ответы в начале разделов)
Что должно быть
Каждый H2 раздел должен следовать структуре:
      1. H2 заголовок сформулирован как вопрос или проблема
      2. Первое предложение после H2 = прямой ответ на этот вопрос
      3. Ответ выделен жирным шрифтом (<strong> или <b>)
      4. После ответа идут детали, примеры, объяснения
Пример ПРАВИЛЬНОЙ структуры
xml
<h2>Сколько можно заработать на Wildberries?</h2>


<p><strong>Средний заработок на Wildberries: 50-150k₽/месяц для новичков, 
300-500k₽/месяц для опытных селлеров.</strong> Доход зависит от категории товара, 
маржи и объёма продаж.</p>


<p>Детальная разбивка по категориям:</p>
<ul>
  <li>Одежда (дропшиппинг): 50-80k₽/мес, маржа 40%</li>
  <li>Электроника (арбитраж): 100-200k₽/мес, маржа 25%</li>
  <li>Собственный бренд: 300-1000k₽/мес, маржа 60%</li>
</ul>


<p>Пример из практики: Анна из Екатеринбурга начала с дропшиппинга одежды 
в январе 2025, стартовый бюджет 25,000₽. Через 3 месяца вышла на 
80,000₽/месяц выручки, чистая прибыль 28,000₽ (35% маржа).</p>


Почему это правильно:
      * ✅ H2 = вопрос
      * ✅ Первое предложение = конкретный ответ с цифрами
      * ✅ Ответ выделен жирным
      * ✅ Далее детали и пример
Пример НЕПРАВИЛЬНОЙ структуры (так НЕ НАДО)

✅ 2.8 Изображения (Images)
Что должно быть
      * Featured image (главная картинка) в начале статьи
      * Минимум 2-3 дополнительных изображения по тексту статьи
      * Каждая картинка имеет:
      * alt атрибут (описание для SEO и слабовидящих)
      * width и height атрибуты (для Core Web Vitals)
      * loading="lazy" (для ускорения загрузки страницы)
      * Оптимизированный размер файла (< 200KB)

  height="600"
  loading="lazy"
  decoding="async"
/>

      * ✅ Описательный alt text (не пустой, не "img1")
      * ✅ Явно указаны width и height
      * ✅ Есть loading="lazy"

✅ 2.9 Объём контента (Word Count)
Что должно быть
      * Минимум 2000 слов на статью
      * Оптимально: 2500-3000 слов
      * Максимум: не более 4000 слов (иначе слишком длинно)
Как проверить
Способ 1: Онлайн word counter (РЕКОМЕНДУЕТСЯ)
      1. Открыть статью на сайте
      2. Выделить весь текст статьи (Ctrl+A)
      3. Скопировать (Ctrl+C)
      4. Перейти на сайт: https://wordcounter.net/
      5. Вставить текст (Ctrl+V)
      6. Посмотреть результат в поле "Words"
Способ 2: Google Docs
      1. Открыть статью на сайте
      2. Скопировать весь текст (Ctrl+A → Ctrl+C)
      3. Создать новый документ в Google Docs
      4. Вставить текст (Ctrl+V)
      5. Меню → Tools → Word count
      6. Посмотреть количество слов
Способ 3: Microsoft Word
      1. Скопировать текст статьи
      2. Вставить в Word
      3. Внизу окна Word автоматически показывает word count
      4. Или: Review → Word Count

БЛОК 5: ПРОВЕРКА ПОСЛЕ ПУБЛИКАЦИИ (Post-Launch)
✅ 5.1 Проверка индексации в Google
Когда проверять
Через 24-48 часов после публикации статьи.
Как проверить
Способ 1: Поиск в Google
      1. Открыть Google.com
      2. Ввести в поиск: site:example.com заголовок статьи
      * Где example.com — ваш домен
      * Где заголовок статьи — H1 вашей статьи
      3. Нажать Enter
      4. Результат: Статья должна быть найдена в первых результатах
Способ 2: Google Search Console (РЕКОМЕНДУЕТСЯ)
      1. Перейти на 
      2. https://search.google.com/search-console
      3. Выбрать ваш сайт
      4. Меню слева → URL Inspection (Проверка URL)
      5. Вставить полный URL статьи
      6. Нажать Enter
      7. Посмотреть статус:
      * ✅ "URL is on Google" (зелёная галочка) — индексирована
      * ⏳ "URL is not on Google" — ещё не проиндексирована
      * ❌ "URL has issues" — есть проблемы
Если статья не проиндексирована через 48 часов:
      1. В URL Inspection нажать "Request Indexing" (Запросить индексацию)
      2. Подождать 24 часа
      3. Проверить снова

      3. Ввести главное ключевое слово
      4. Прокрутить результаты поиска
      5. Искать вашу статью на первых 3-5 страницах (топ-30 — топ-50)
      6. Записать позицию

      * https://serpstat.com

🔗 КАК ОНИ СВЯЗАНЫ (МЕХАНИКА ВЗАИМОДЕЙСТВИЯ)
Основной принцип: Мастер (WF 0) вызывает остальные воркфлоу через ноду Execute Workflow и ждет ответа.
         1. Запуск: WF 0 берет строку из Google Sheets.
         2. Планирование: WF 0 вызывает WF 1. Получает назад JSON Blueprint.
         3. Параллельное производство: WF 0 одновременно запускает WF 2 и WF 3.
         * Внутри WF 2 и WF 3, когда нужно нарисовать картинку, они вызывают WF 7 (Image Gen) и ждут ссылку.
         4. Ожидание и Склейка: WF 0 ждет, пока придут ответы от 2 и 3, и передает всё в WF 4 (Assembly).
         5. Публикация: Готовый HTML из WF 4 летит в WF 5.
         6. Фоновые задачи: Как только статья на сайте, WF 0 запускает WF 6 (Indexer) и не ждет его (пусть гуглит в фоне), а сам сразу закрывает статус в таблице на Done.

🧐 ЗАМЕТКИ КОМАНДЫ ДЛЯ РАЗРАБОТЧИКА
         * Изоляция (WF 7): Это самое важное решение. Генерация картинки может длиться 2 минуты. Если делать это внутри тела, воркфлоу №3 «отвалится» по таймауту. WF 7 работает автономно и имеет право на «ошибку и повтор».
         * Контекст (Payload): Мы договорились передавать между воркфлоу не просто текст, а объект ArticleContext. Это гарантирует, что Журналист в WF 4 «помнит», что Архитектор в WF 1 заложил в статью.
         * LSI и Досье: Эти данные собираются один раз в WF 1 и протаскиваются через Мастера до самого конца.
Это отличный вопрос, который раскрывает суть сервисной архитектуры. WF №7 — это «удаленный цех» по покраске. Он не работает сам по себе, его вызывают те воркфлоу, которым в данный момент «понадобилась краска» (картинка).
WF №7 является подчиненным (Sub-workflow). Он запускается по запросу из двух точек производства.

🗺️ Схема связей WF №7 (IMAGE GEN)
1. Кто и когда его запускает?
Его вызывают два «производственных цеха», где сидит Агент №8 (Арт-директор):
         * Точка А: Воркфлоу №2 (HOOK — Голова статьи)
         * Когда: Сразу после того, как Журналист написал H1, а Арт-директор составил промпт для главной обложки.
         * Зачем: Чтобы получить Featured Image (главное фото статьи).
         * Точка Б: Воркфлоу №3 (BODY — Тело статьи)
         * Когда: Внутри цикла (Loop), когда робот натыкается на блок type: "IMG".
         * Зачем: Чтобы получить Inline Image (картинку внутри текста для иллюстрации конкретного раздела).

🔄 Механика вызова (Step-by-Step)
Разработчик настраивает это через ноду Execute Workflow. Вот как это выглядит изнутри:
         1. Запрос (Call): WF №2 или №3 отправляет в WF №7 «посылку» (JSON).
         * Состав посылки:
         * prompt: Текст на английском от Арт-директора.
         * model_id: Какую модель использовать (из SETTINGS — Flux или MJ).
         * ref_url: Ссылка на фото героя (из SETTINGS).
         * ref_mode: Character или Style (из SETTINGS).
         2. Остановка (Wait): Вызывающий воркфлоу (например, Тело статьи) замирает. Он не идет к следующему блоку, пока WF №7 не вернет ответ.
         3. Работа WF №7: * Стучится в API Kie.ai.
         * Если API занято — ждет и пробует снова (Retry Logic).
         * Получает финальную ссылку на картинку.
         4. Ответ (Return): WF №7 отдает одну единственную строку: image_url.
         5. Продолжение: WF №2 или №3 «просыпается», забирает ссылку и вставляет её в HTML-код.

🛠️ Почему это выгодно? (Пример сбоя)
Представь, что ты генерируешь статью про ПВЗ. Робот дошел до середины, и в этот момент Midjourney «упал» на 2 минуты.
         * Если бы картинка была внутри WF №3: Весь твой огромный воркфлоу с текстами, ключами и логикой вылетел бы с ошибкой. Тебе пришлось бы перезапускать всё и тратить токены заново.
         * С отдельным WF №7: Основной воркфлоу №3 просто «висит» и ждет. А WF №7 внутри себя спокойно делает 3–4 попытки достучаться до нейросети. Как только она «ожила», он отдал картинку, и основная сборка пошла дальше как ни в чем не бывало.

Нода 1.4: Code (Geo & Language Logic) — МОЗГ ЭТАПА
Здесь происходит техническое распределение. Нода анализирует текст запроса и страну, чтобы выдать правильные ID для API.
            * Логика работы:
            1. Cyrillic Test: Проверяет, написан ли ключ кириллицей.
            2. Yandex vs Google: Если есть кириллица ИЛИ страна «Россия», выбирается режим Яндекса.
            3. Location ID: Находит числовой код региона в словаре.







Финальный код для копирования в n8n:
JavaScript

const item = $input.item.json;
const keyword = item["Main Keyword"];
const country = item["Target Country"] || "";
const city = (item["Target City"] || "").toLowerCase().trim();


// Словари ID регионов
const yandexRegions = { "москва": 213, "спб": 2, "default": 225 };
const googleLocations = { "казахстан": 2398, "беларусь": 2112, "сша": 2840, "default": 2840 };


let mode = "google_dataforseo";
let locationCode = googleLocations["default"];
let languageCode = "en";


// Проверка языка и страны
const hasCyrillic = /[а-яА-ЯёЁ]/.test(keyword);
const isRussia = country.includes("Россия") || (country === "" && hasCyrillic);


if (isRussia) {
    mode = "yandex_xmlriver";
    locationCode = yandexRegions[city] || yandexRegions["default"];
    languageCode = "ru";
} else if (country.includes("Казахстан") || country.includes("Беларусь")) {
    locationCode = country.includes("Казахстан") ? googleLocations["казахстан"] : googleLocations["беларусь"];
    languageCode = "ru";
}


return { json: { mode, locationCode, languageCode, keyword } };

🧬 ЭТАП 2: СБОР SEO И ФАКТОВ (ПАРАЛЛЕЛЬНО)

Нода 2.2: Code (SEO Parser & Cleaner)
            * Вход: Сырые данные от API.
            * Логика (JavaScript):
            * Фильтрация: Оставляет только запросы с объемом (volume) > 50.
            * Чистка мусора: Удаляет фразы со стоп-словами («скачать», «торрент», «бесплатно», «порно» и т.д.).
            * Дедупликация: Удаляет главный ключ и повторяющиеся фразы.
            * Лимит: Оставляет строго Топ-20 самых частотных LSI-ключей.
            * Выход: Чистый массив lsi_keywords.
Это «промышленный измельчитель». Нода принимает сырые данные от XMLRiver или DataForSEO и превращает их в массив из 20 идеальных LSI-ключей.
⚙️ Подробные правила фильтрации и чистки:
            1. Порог объема (Volume Threshold): Игнорируем «мусорные» запросы с частотностью менее 50 показов в месяц, чтобы не тратить внимание нейросети на низкочастотный шум.
            2. Черный список слов (Stop-word List):
            * Информационный мусор: «скачать», «торрент», «бесплатно», «реферат», «курсовая», «порно», «ххх», «смотреть», «онлайн».
            * Навигационный мусор (чужие площадки): «авито», «ютуб», «youtube», «вк», «озон», «wildberries» (если тема не про них), «форум», «отзывы».
            * Временной мусор: «2022», «2023», «2024», «прошлый год» (мы работаем только в контексте 2026).
            3. Интеллектуальная дедупликация:
            * Удаление точного дубля Main Keyword.
            * Удаление фраз, которые короче 3 символов.
            * Нормализация (приведение к нижнему регистру, удаление лишних пробелов).
            4. Сортировка и Лимит: Ранжирование по убыванию частотности и выборка строго Топ-20.
💻 Итоговый JavaScript код для ноды n8n:
JavaScript

// Получаем данные от предыдущей ноды (Яндекс или Google)
const input = $input.item.json;
const mainKey = $("Code (Geo Logic)").item.json.keyword.toLowerCase().trim();
let rawItems = [];


// Автоматическое определение структуры входящих данных
if (input.content && input.content.items) {
    // Данные от XMLRiver (Яндекс)
    rawItems = input.content.items.map(i => ({ word: i.phrase.toLowerCase().trim(), vol: parseInt(i.shows) }));
} else if (input.tasks && input.tasks[0].result) {
    // Данные от DataForSEO (Google)
    rawItems = input.tasks[0].result[0].items.map(i => ({ word: i.keyword.toLowerCase().trim(), vol: i.search_volume }));
}


// РАСШИРЕННЫЙ СПИСОК СТОП-СЛОВ
const stopWords = [
    "скачать", "бесплатно", "торрент", "реферат", "курсовая", "порно", "ххх", 
    "смотреть", "онлайн", "видео", "ютуб", "youtube", "вк", "авито", "форум", 
    "отзывы", "2021", "2022", "2023", "2024", "2025", "б/у", "авито", "олх"
];


const cleanKeywords = rawItems
    .filter(item => item.vol > 50) // Правило 1: Частотность
    .filter(item => {
        // Правило 2: Чистка мусора
        return !stopWords.some(stop => item.word.includes(stop));
    })
    .filter(item => {
        // Правило 3: Дедупликация главного ключа
        return item.word !== mainKey && item.word.length > 3;
    })
    .sort((a, b) => b.vol - a.vol) // Правило 4: Сортировка по весу
    .slice(0, 20) // Лимит: ТОП-20
    .map(item => item.word);


return { json: { lsi_keywords: cleanKeywords } };

Нода 4.1: Code Validator (Главный контролер)
Это нода типа Code (JavaScript), которая анализирует JSON-ответ от Claude 3.5 Sonnet.
⚙️ Подробные настройки и логика (Script):

// 1. ПОЛУЧЕНИЕ ДАННЫХ
const content = $input.item.json.choices[0].message.content; // Ответ от Claude
let plan;
let errors = [];


// 2. УРОВЕНЬ: СИНТАКСИС (JSON Syntax)
try {
    const cleanJson = content.replace(/```json\n?|```/g, "").trim();
    plan = JSON.parse(cleanJson);
} catch (e) {
    errors.push("CRITICAL: Invalid JSON syntax. Claude output bad code.");
}


if (plan) {
    // 3. УРОВЕНЬ: СТРУКТУРА (Бизнес-логика COMANDOS AI)
    // Проверка количества блоков (Должно быть 13-16)
    if (!Array.isArray(plan) || plan.length < 13 || plan.length > 16) {
        errors.push(`ОШИБКА: Количество блоков (${plan.length}) вне диапазона 13-16.`);
    }


    // Проверка обязательных типов блоков
    const types = plan.map(b => b.type);
    const requiredTypes = {
        'TABLE': { min: 1, max: 2 },
        'RESEARCH': { min: 1 },
        'STEPS': { min: 1 },
        'QUOTE': { min: 1 },
        'IMG': { min: 3 },
        'SOFT_OFFER': { min: 1 }
    };


    for (const [type, rule] of Object.entries(requiredTypes)) {
        const count = types.filter(t => t === type).length;
        if (rule.min && count < rule.min) errors.push(`ОШИБКА: Недостаточно блоков типа ${type} (нужно min ${rule.min}).`);
        if (rule.max && count > rule.max) errors.push(`ОШИБКА: Слишком много блоков типа ${type} (нужно max ${rule.max}).`);
    }


    // 4. УРОВЕНЬ: SEO (Наличие ключей)
    const hasKeywords = plan.some(b => b.keywords && b.keywords.length > 0);
    if (!hasKeywords) errors.push("ОШИБКА: В блоках отсутствуют LSI-ключи.");
}


// 5. ВЫДАЧА РЕЗУЛЬТАТА
return { 
    json: { 
        isValid: errors.length === 0, 
        errors: errors, 
        plan: plan || null,
        retryCount: $node["GS: Trigger New Task"].json.retryCount || 0 // Счетчик попыток
    } 
};
